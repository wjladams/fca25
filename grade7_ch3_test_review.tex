    \begin{enumerate}
        \item $7x-y+3y-6x$

        \item $4x + 3 - 5x + 5$
    \end{enumerate}
    \item Expand the following:
    \begin{enumerate}
        \item $3(a+2)$
        \item $a(3a+5)$
        \item $(x+5)3$
        \item $-2(a-5)$
    \end{enumerate}
